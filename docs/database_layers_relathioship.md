### **2. Service Layer Boundaries**
```python
# âœ… CORRECT: Service orchestrates both models properly
class UserService:
    def create_user(self, user_in: DomainUserCreate) -> DomainUser:
        # 1. Business logic with domain model
        domain_user = DomainUser.create(user_in.email, user_in.name)
        # 2. Convert to database model
        db_user = DBUser.from_domain(domain_user)
        # 3. Database operation
        saved_db_user = self.user_repo.create(db_user)
        # 4. Convert back to domain model for business context
        return saved_db_user.to_domain()

# âŒ ANTI-PATTERN: Mixing concerns
class UserService:
    def create_user(self, user_in: dict) -> dict:  # No type safety
        # Business logic and database operations mixed
        if "@" not in user_in["email"]:
            raise ValueError("Invalid email")  # Business rule in service
        db_user = DBUser(**user_in)  # Direct database model creation
        self.db.add(db_user)  # Direct database access
        self.db.commit()
        return db_user  # Returns database model
```
